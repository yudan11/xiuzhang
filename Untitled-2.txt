float calculate(int i,float a,float b)
{
 float x;
 switch(i)
 {
 case 0:
  x=add(a,b);
  break;
 case 1:
  x=subtraction(a,b);
  break;
 case 2:
  x=multiplication(a,b);
  break;
 case 3:
  x=division(a,b);
  break;
 default:
  break;
 }
 return x;
}

 return x;
}


float division(float a,float b)
{
 float x;
 x=a/b;
 return x;
}



char symbol(int i)
{
 char op;
 switch(i)
 {
 case 0:
  op='+';
  break;
 case 1:
  op='-';
  break;
 case 2:
  op='*';
  break;
 default:
  op='/';
  break;
 }
 return op;
}
